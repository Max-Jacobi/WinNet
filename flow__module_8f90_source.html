201 !>
202 !! Flow calculation from jacobian. It is calculated with the help of the Jacobian.
203 !! \f[
204 !! F_{ij} = (J_{ji} \times Y_j - J_{ij} \times Y_i) \Delta t
205 !! \f]
206 !!
207 !! At each timestep the flows are cumulatively added to the flows and/or cum_flows arrays.
208 !! On each output iteration the flows array and the cumulative timestep is reset to zero.
209 !!
210 !! @note Using the jacobian directly has the advantage
211 !! that the flow will be correct if the calculation
212 !! is correct. In previous versions, the flow
213 !! was not calculated by using the jacobian.
214 !!
215 !! \b Edited:
216 !! - 03.04.18, M. Jacobi
217 !! - 25.08.23, M. Jacobi cumulativ flow integration
218 !! .
219 subroutine flowcalc(Y, dt)
220  implicit none
221 
222  ! MJ: these could in principle be used from single_zone_vars
223  real(r_kind), dimension(:), intent(in) :: y !< abundances
224  real(r_kind), intent(in) :: dt !< time step
225 
226  real(r_kind) :: fl
227  integer :: i, j, ij, ji, n
228 
229  ! update cumulative time step
230  cum_dt = cum_dt + dt
231 
232  do n = 1, flow_size
233  if (mode.eq.1) then
234  i = flows(n)%i
235  j = flows(n)%j
236  ij = flows(n)%ij
237  ji = flows(n)%ji
238  else
239  i = cum_flows(n)%i
240  j = cum_flows(n)%j
241  ij = cum_flows(n)%ij
242  ji = cum_flows(n)%ji
243  end if
244 
245  ! add up momentary flows in temporary variable
246  fl = 0
247  if (ij.ne.0) fl = - vals(ij) * y(i) * dt
248  if (ji.ne.0) fl = fl + vals(ji) * y(j) * dt
249 
250  ! add flow only if it is not abnormally large
251  ! (can sometimnes happen in the beginning)
252  if (abs(fl) < 1e99) then
253  select case (mode)
254  case (1)
255  flows(n)%fl = flows(n)%fl + fl
256  case (2)
257  cum_flows(n)%fl = cum_flows(n)%fl + fl
258  case (3)
259  flows(n)%fl = flows(n)%fl + fl
260  cum_flows(n)%fl = cum_flows(n)%fl + fl
261  end select
262  endif
263  end do
264 
265  info_exit("flowcalc")
266 
267 end subroutine flowcalc
